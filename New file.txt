# Simple Food Rescue Map App

Here's a simple Food Rescue Map activity and layout that displays food rescue locations on a map:

```kotlin name=MainActivity.kt
package com.example.foodrescue

import android.os.Bundle
import androidx.appcompat.app.AppCompatActivity
import org.osmdroid.config.Configuration
import org.osmdroid.tileprovider.tilesource.TileSourceFactory
import org.osmdroid.util.GeoPoint
import org.osmdroid.views.MapView
import org.osmdroid.views.overlay.Marker

class MainActivity : AppCompatActivity() {

    private lateinit var mapView: MapView

    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        
        // Load OSMDroid configuration
        Configuration.getInstance().load(applicationContext, getSharedPreferences("osmdroid", MODE_PRIVATE))
        
        setContentView(R.layout.activity_main)

        // Initialize map view
        mapView = findViewById(R.id.mapView)
        mapView.setTileSource(TileSourceFactory.MAPNIK)
        mapView.setMultiTouchControls(true)
        
        // Set initial position and zoom level
        val mapController = mapView.controller
        mapController.setZoom(15.0)
        
        // Use Chennai as the center point
        val centerPoint = GeoPoint(13.0827, 80.2707) // Chennai
        mapController.setCenter(centerPoint)
        
        // Add food rescue locations to map
        addFoodRescueLocations()
    }
    
    private fun addFoodRescueLocations() {
        // Define food rescue locations
        val locations = listOf(
            Pair("Green Leaf Restaurant", GeoPoint(13.0830, 80.2715)),
            Pair("Daily Bread Bakery", GeoPoint(13.0810, 80.2695)),
            Pair("Spice Corner", GeoPoint(13.0805, 80.2720)),
            Pair("City Mart", GeoPoint(13.0790, 80.2740)),
            Pair("Fresh Farms Market", GeoPoint(13.0850, 80.2690)),
            Pair("Community Food Bank", GeoPoint(13.0820, 80.2670)),
            Pair("Hope Food Pantry", GeoPoint(13.0860, 80.2730))
        )
        
        // Add markers for each location
        locations.forEach { (name, point) ->
            val marker = Marker(mapView)
            marker.position = point
            marker.setAnchor(Marker.ANCHOR_CENTER, Marker.ANCHOR_BOTTOM)
            marker.title = name
            mapView.overlays.add(marker)
        }
        
        // Refresh the map
        mapView.invalidate()
    }

    override fun onResume() {
        super.onResume()
        mapView.onResume()
    }

    override fun onPause() {
        super.onPause()
        mapView.onPause()
    }
}
```


## Implementation Notes:

1. **Map Implementation**:
   - Uses OSMDroid library for displaying the map
   - Shows Chennai as the default location (coordinates 13.0827, 80.2707)

2. **Food Rescue Locations**:
   - Displays 7 food rescue locations as markers on the map
   - Locations include restaurants, grocery stores, and food banks

3. **Library Requirement**:
   Add to your app's build.gradle:
   ```gradle
   implementation 'org.osmdroid:osmdroid-android:6.1.13'
   ```

4. **Permissions**:
   Add to your AndroidManifest.xml:
   ```xml
   <uses-permission android:name="android.permission.INTERNET" />
   <uses-permission android:name="android.permission.ACCESS_NETWORK_STATE" />
   ```

This implementation provides a simple single-page application that displays food rescue locations on a map centered on Chennai.
